---
title: "Get Unit"
api: "GET https://api.travtus.com/ingestion/unit/{externalref}/"
description: "This endpoint retrieves infromation about one unit with the provided external reference."
---

### Header

<ParamField header="Auhtorization" type="string" default="none" required>
  The authentication token for your request.
</ParamField>

### Path Parameters

<ParamField path="externalref" type="string" default="none" required>
  The external reference for the unit to be retrieved.
</ParamField>

### Response

<ResponseField name="identifier" type="string">
  The internal identifier for the retrieved unit. Unique across all portfolios.
</ResponseField>

<ResponseField name="unit_number" type="string">
  The number of the retrieved unit.
</ResponseField>

<ResponseField name="is_flex" type="boolean">
  True if the retrieved unit is a flex apartment, false if it is not.
</ResponseField>

<ResponseField name="full_bathrooms" type="integer">
  The number of full bathrooms for the retrieved unit.
</ResponseField>

<ResponseField name="half_bathrooms" type="integer">
  The number of half bathrooms for the retrieved unit.
</ResponseField>

<ResponseField name="number_of_bedrooms" type="integer">
  The number of bedrooms for the retrieved unit.
</ResponseField>

<ResponseField name="sqft" type="integer">
  The square feet surface for the retrieved unit.
</ResponseField>

<ResponseField name="building" type="object">
  The building record associated with the unit.

  <Expandable title="building">
    <ResponseField name="building_identifier" type="string">
      The internal identifier for the unit's building. Unique across all portfolios.
    </ResponseField>
    <ResponseField name="address" type="object">
      The address of the retrieved unit's building.
      <Expandable tilte="address">
        <ResponseField name="building_name" type="string">
          The name of the building.
        </ResponseField>
        <ResponseField name="address_line_1" type="string">
          The first line of the building's address.
        </ResponseField>
        <ResponseField name="address_line_2" type="string">
          The second line of the building's address.
        </ResponseField>
        <ResponseField name="city" type="string">
          The city of the building's address.
        </ResponseField>
        <ResponseField name="country" type="string">
          The country of the building's address.
        </ResponseField>
        <ResponseField name="state" type="string">
          The state in which the building's address is located.
        </ResponseField>
        <ResponseField name="postal_code" type="string">
          The postal code of the building's address.
        </ResponseField>
        <ResponseField name="latitude" type="string">
          The latitude of the building's address.
        </ResponseField>
        <ResponseField name="longitude" type="string">
          The longitude of the building's address.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>

</ResponseField>
